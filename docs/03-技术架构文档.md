## 一、架构风格

**单机静态分析器** — 纯文件扫描，零网络，零状态。

```
┌────────────────────────────────────────────────────────────┐
│                    ai-lint CLI 入口                         │
│  ai-lint │ ai-lint fix │ ai-lint --cross-files             │
└──────┬─────────────────────────────────────────────────────┘
       │
┌──────▼─────────────────────────────────────────────────────┐
│                    Rule Engine                              │
│  ┌──────────────┐ ┌──────────────┐ ┌────────────────────┐  │
│  │ Parser       │ │ Rule Runner  │ │ Fixer              │  │
│  │ Markdown→AST │ │ 规则检查流水线│ │ 自动修复引擎       │  │
│  └──────┬───────┘ └──────┬───────┘ └────────┬───────────┘  │
│         │                │                   │              │
│  ┌──────▼────────────────▼───────────────────▼───────────┐  │
│  │                  12 条检测规则                          │  │
│  │  no-duplicate │ no-conflict │ no-overconstrain │ ...  │  │
│  └───────────────────────────────────────────────────────┘  │
└──────┬─────────────────────────────────────────────────────┘
       │
┌──────▼─────────────────────────────────────────────────────┐
│                    本地文件系统                              │
│  CLAUDE.md / AGENTS.md / skills/*/SKILL.md / .cursorrules  │
└────────────────────────────────────────────────────────────┘
```

## 二、技术栈

### 生产依赖（4 个）

| 依赖 | 用途 |
|------|------|
| `chalk` | 终端着色 |
| `commander` | CLI 框架 |
| `fs-extra` | 文件操作 |
| `pathe` | 路径处理 |

### 明确不用的

| 不用 | 原因 |
|------|------|
| boxen | 输出格式不需要框线，简洁列表格式更接近 ESLint |
| ora | 扫描速度 < 1s，不需要 loading 动画 |
| execa | 纯文件扫描，不需要进程执行 |
| smol-toml | 目标文件全是 Markdown/YAML frontmatter，不需要 TOML |
| Ink | React 渲染引擎过重 |
| i18next | MVP 只支持中文 |

## 四、核心模块设计

### 4.1 Rule 接口

```typescript
/** 问题严重级别 */
type Severity = 'error' | 'warning'

/** 一条检测结果 */
interface LintIssue {
  ruleId: string           // 'no-duplicate'
  severity: Severity
  file: string             // 'CLAUDE.md'
  line?: number            // 行号
  column?: number
  message: string          // 人类可读的描述
  tokenWaste?: number      // 估算的 Token 浪费量
  fixable: boolean         // 是否可自动修复
}

/** 规则接口 */
interface Rule {
  id: string
  description: string
  /** 适用文件类型 */
  files: string[]          // ['CLAUDE.md', 'AGENTS.md', 'SKILL.md', ...]
  /** 检查函数 */
  check(content: string, filePath: string): LintIssue[]
  /** 修复函数（可选） */
  fix?(content: string, issue: LintIssue): string
}
```

### 4.2 Parser — Markdown 解析

```typescript
/**
 * 将 Markdown 文本解析为段落列表。
 * 每条规则 = 一个列表项或独立段落。
 *
 * 输入:
 *   "- 使用 TypeScript 严格模式"
 *   "- 所有函数必须添加 JSDoc 注释"
 *   "- 使用 TypeScript 严格模式，启用 noUncheckedIndexedAccess"
 *
 * 输出:
 *   [
 *     { text: "使用 TypeScript 严格模式", line: 2 },
 *     { text: "所有函数必须添加 JSDoc 注释", line: 3 },
 *     { text: "使用 TypeScript 严格模式，启用 noUncheckedIndexedAccess", line: 4 },
 *   ]
 */
function parseRules(content: string): RuleEntry[]
```

### 4.3 Parser — YAML Frontmatter

```typescript
/**
 * 从 SKILL.md 中提取 YAML frontmatter。
 *
 * 输入:
 *   ---
 *   name: brandkit
 *   description: Premium brand-kit image generation skill...
 *   ---
 *   # BRANDKIT IMAGE GENERATION SKILL
 *   ...
 *
 * 输出:
 *   { name: "brandkit", description: "Premium brand-kit..." }
 */
function parseFrontmatter(content: string): SkillMeta | null
```

### 4.4 no-duplicate（核心规则示例）

```typescript
export const noDuplicate: Rule = {
  id: 'no-duplicate',
  description: '检测字面重复的规则',
  files: ['CLAUDE.md', 'AGENTS.md', 'SKILL.md', '.cursorrules'],
  check(content: string, filePath: string): LintIssue[] {
    const rules = parseRules(content)
    const seen = new Map<string, number[]>()
    const issues: LintIssue[] = []

    for (const rule of rules) {
      const normalized = rule.text.trim().toLowerCase()
      if (seen.has(normalized)) {
        seen.get(normalized)!.push(rule.line)
      } else {
        seen.set(normalized, [rule.line])
      }
    }

    for (const [text, lines] of seen) {
      if (lines.length > 1) {
        issues.push({
          ruleId: 'no-duplicate',
          severity: 'error',
          file: filePath,
          line: lines[1], // 第二次出现的位置
          message: `"${text.slice(0, 60)}" 出现 ${lines.length} 次 (行 ${lines.join(', ')})`,
          tokenWaste: estimateTokenWaste(text, lines.length),
          fixable: true,
        })
      }
    }
    return issues
  },
  fix(content: string, issue: LintIssue): string {
    // 移除重复行，保留第一次出现
    return removeDuplicateLines(content, issue)
  },
}
```

### 4.5 no-overlap-skills（跨文件规则示例）

```typescript
export const noOverlapSkills: CrossFileRule = {
  id: 'no-overlap-skills',
  description: '检测两个 Skill 的触发域或指令高度重叠',
  check(skills: Map<string, SkillMeta>): LintIssue[] {
    const issues: LintIssue[] = []
    const entries = [...skills.entries()]

    for (let i = 0; i < entries.length; i++) {
      for (let j = i + 1; j < entries.length; j++) {
        const [nameA, metaA] = entries[i]
        const [nameB, metaB] = entries[j]
        const overlap = calculateDescriptionOverlap(metaA.description, metaB.description)

        if (overlap > 0.7) {
          issues.push({
            ruleId: 'no-overlap-skills',
            severity: 'warning',
            file: `${nameA}/SKILL.md ↔ ${nameB}/SKILL.md`,
            message: `description 重叠度: ${Math.round(overlap * 100)}%`,
            fixable: false,
          })
        }
      }
    }
    return issues
  },
}
```
